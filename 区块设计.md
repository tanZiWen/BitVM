pub struct BlockHeader { 
  prev_block_hash: Hash, // 前一区块哈希 
  merkle_root: Hash, // 交易的 Merkle 树根 
  utxo_root: Hash,
  metadata: 
  proof : proof
}

// 区块结构体 #[derive(Serialize, Deserialize, Clone, Debug)] 
pub struct Block { 
  pub header: BlockHeader, // 区块头 
  pub transactions: Vec, // 交易列表 
}
