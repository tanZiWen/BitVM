fn validate_transaction(tx: &Transaction, circuit_data: &CircuitData<F, C, D>) -> Result<(), String> {
    // 检查基本字段
    if tx.amount < F::ZERO || tx.gasprice < F::ZERO {
        return Err("Invalid amount or gasprice".to_string());
    }

    // 获取账户状态
    let account = get_account_state(tx.from).ok_or("Account not found")?;

    // 验证 nonce
    if tx.nonce != account.nonce + F::ONE {
        return Err("Invalid nonce".to_string());
    }

    // 验证余额
    let gas_cost = tx.gasprice * F::from_canonical_u64(21000); // 假设 gas_limit = 21000
    let total_cost = tx.amount + gas_cost;
    if total_cost > account.balance {
        return Err("Insufficient balance".to_string());
    }

    // 验证 Plonky2 proof证明
    if verify::<F, C, D>(circuit_data, &tx.proof, &public_inputs).is_err() {
        return Err("Invalid proof".to_string());
    }

    Ok(())
}
